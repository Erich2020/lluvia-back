Create Database Temporada;
use Temporada;
Create Table Tickets (
id bigint not null autoincrement,
listaproductos varchar(9999),
total Decimal(8,2),
cantidad Decimal(8,2),
fecha varchar(15),
hora varchar(15),
fk_usuario int references usuario(id)
);
Create Table Productos(
codigo varchar(60) not null,
descripcion varchar(9999),
costo Decimal(8,2),
precio Decimal(8,2),
existencia Decimal (8,2)
);
Create Table Usuarios(
Id int not null autoincrement,
nombre varchar(200),
username varchar(20),
password varchar(70)
rol varchar(20)
);
